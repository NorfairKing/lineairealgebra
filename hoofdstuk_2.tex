\subsection*{oef 3}
\begin{center}
$det(AB) = 0$ als $det(A) = 0$ of $det(B) = 0$
\\
\[det(A) = \begin{vmatrix}
x + 2 & 3x\\
3 & x + 2
\end{vmatrix}
 = (x + 2)^2 - 9x = x^2 -5 x + 4
\]
$x^2 -5 x + 4 = 0$ voor $x_1, x_2$ met $x_{1,2} = \frac{5 \pm 3}{2}$ ($x_1 = 1$ en $x_2 = 4$)
\\
\[det(B) = \begin{vmatrix}
x & 0\\
5 & x + 2
\end{vmatrix}
 = x \cdot (x + 2) = x^2 + 2x\]
$x^2 + 2x$ voor $x_3, x_4$ met $x_3 = -2$ en $x_4 = 0$
\end{center}
\subsubsection*{Besluit}
$det(AB) = 0$ voor $x$ met $x \in \{-2, 0, 1, 4\}$.
